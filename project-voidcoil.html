Centralized audio (SFX + music) C++ / Gateware Audio 

	void PlaySoundEffect(entt::registry& registry, std::string iniName)
	{
		//Get the sound effects component
		auto view = registry.view<GAME::SoundEffects>();
		auto& sfx = registry.get<GAME::SoundEffects>(view[0]).effects;

		//Check if the current effect has already been created - if not create it
		if (sfx.find(iniName) == sfx.end())
		{
			auto& config = registry.ctx().get<UTIL::Config>().gameConfig;
			GW::AUDIO::GAudio& gAudio = registry.ctx().get<GW::AUDIO::GAudio>();
			sfx.emplace(iniName, GW::AUDIO::GSound());
			sfx[iniName].Create((*config).at(iniName).at("soundPath").as<std::string>().c_str(), gAudio);
		}

		//Play sound
		auto iter = sfx.find(iniName);
		iter->second.Play();
	}

	void PlayMusic(entt::registry& registry, std::string iniName)
	{
		auto view = registry.view<GAME::BackgroundMusic>();
		auto& musicTracks = registry.get<GAME::BackgroundMusic>(view[0]).musicTracks;
		bool playing;

		//Check if the current track exists first - if not add it 
		if (musicTracks.find(iniName) == musicTracks.end())
		{
			auto& config = registry.ctx().get<UTIL::Config>().gameConfig;
			GW::AUDIO::GAudio& gAudio = registry.ctx().get<GW::AUDIO::GAudio>();
			musicTracks.emplace(iniName, GW::AUDIO::GMusic());
			musicTracks[iniName].Create((*config).at(iniName).at("musicPath").as<std::string>().c_str(), gAudio);
		}

		//Loop through each track and stop the tracks not for current level
		for (auto track : musicTracks)
		{
			track.second.isPlaying(playing);

			if (track.first == iniName)
			{
				if (!playing)
					track.second.Play(true);
			}
			else if (playing)
			{
				track.second.Stop();
			}
		}
	}
Controller vibration C++ / Gamepad 

	void VibrateController(entt::registry& registry)
	{
		auto& input = registry.ctx().get<UTIL::Input>();
		bool connected = false;
		input.gamePads.IsConnected(0, connected);
		if (connected)
		{
			auto& view = registry.view<GAME::ControllerVibration>();
			auto vibration = registry.get<GAME::ControllerVibration>(view[0]);
			if (vibration.vibrate)
			{
				input.gamePads.StartVibration(0, 0, vibration.duration, vibration.strength);
			}
		}
	}
